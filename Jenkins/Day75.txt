#script to install jenkins to add them as nodes/slaves.

#plugin: build agent with SSH

sshpass -p Ir0nM@n sudo yum install wget -y
sshpass -p Ir0nM@n sudo wget -O /etc/yum.repos.d/jenkins.repo https://pkg.jenkins.io/redhat-stable/jenkins.repo
sshpass -p Ir0nM@n sudo rpm --import https://pkg.jenkins.io/redhat-stable/jenkins.io-2023.key
sshpass -p Ir0nM@n sudo yum upgrade -y
sshpass -p Ir0nM@n sudo yum install fontconfig java-21-openjdk -y
sshpass -p Ir0nM@n sudo yum install jenkins -y
sshpass -p Ir0nM@n sudo systemctl daemon-reload
sshpass -p Ir0nM@n sudo systemctl enable jenkins
sshpass -p Ir0nM@n sudo systemctl start jenkins
sshpass -p Ir0nM@n sudo systemctl status jenkins
